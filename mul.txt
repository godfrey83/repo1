int binaryserch(int arr[],int k,int s,int l)
{
    if(s>l)return -1;
    int mid=(s+l)/2;
    if(arr[mid]==k)return mid;
    if(k>arr[mid])return binaryserch(arr,k,mid+1,l);
    return binaryserch(arr,k,s,mid-1);
}
bool isPrime(int n)
{
    // Corner case
    if (n <= 1)
        return false;
 
    // Check from 2 to square root of n
    for (int i = 2; i <= sqrt(n); i++)
        if (n % i == 0)
            return false; 
 
    return true;
}
